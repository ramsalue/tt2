
\chapter{Nomenclatura}
\begin{itemize}[font=\bfseries]
	\item [ACV] Accidente Cerebrovascular
	\item[IMSS] Instituto Mexicano del Seguro Social
	\item[CNC] Control Numérico por Computadora (\textit{por sus siglas en inglés omputer Numerical Control})
	\item[FBS] Estructura de Desglose Funcional (\textit{por sus siglas en inglés Functional Breakdown Structure})
	\item[IDEF0] Integración de Definición para el Modelado de Funciones (por sus siglas en inglés Integration Definition for Function Modeling)
	\item[GUI] Interfaz Gráfica de Usuario (por sus siglas en inglés Graphical User Interface)
	\item[HMI] Interfaz Humano-Máquina (\textit{por sus siglas en inglés Human-Machine Interface})
	\item[ISO] Organización Internacional de Normalización (por sus siglas en inglés International Organization for Standardization)
\end{itemize}
